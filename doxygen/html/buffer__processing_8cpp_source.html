00716  *      Takes a file read buffer's worth of data, passes it off to

00718  *      them to the word_list vector (they will get coallated with the other
00719  *      words in the caller.
00720  *******************************************************************************

00722 int processWholeBuffer(char *buffer, int buffer_sz, list<char *> &word_list)
00723 {
00724     int chars_processed = 0;
00725     char *buffer_start = buffer;
00726     char *word_found = NULL;

00729     int idx = 0;
00730 
00731     DBG(printf ("Buffer size: %d\n", buffer_sz));
00732     /*
00733      * If very last of buffer looks like a word, back off until we are at the
00734      * beginning of it, and then only process up until that point
00735      */
00736     if (isWordChar(buffer[buffer_sz - 1]))
00737     {
00738         for (idx = buffer_sz - 1; idx >= 0; idx--) 
00739         {
00740             if (isWordChar(buffer[idx]) == FALSE)
00741             {
00742                 buffer_sz_to_process = idx;
00743                 break;
00744             }
00745         } /* end for */
00746     }
00747     while (chars_processed < buffer_sz_to_process)
00748     {
00749         processed_this_round = processBufferForWords (buffer_start,
00750                 (buffer_sz_to_process - chars_processed),
00751                 &word_found);
00752         buffer_start += processed_this_round;
00753         chars_processed += processed_this_round;
00754         DBG(printf ("  %d: bytes proceesed this loop\n", processed_this_round));

00756         if (word_found != NULL)

00758             DBG(printf ("---->Found word: %s\n", word_found));
